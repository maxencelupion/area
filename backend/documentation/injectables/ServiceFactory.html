import { BaseService } from './base-service.service';
import { Injectable, OnModuleInit, Type } from '@nestjs/common';
import { ElementService } from '../elements/element.service';
import { EventEmitter2 } from '@nestjs/event-emitter';
import { ServiceManager } from './service-manager.service';
import { GoogleServiceService } from './google-service.service';
import { MicrosoftServiceService } from './microsoft-service.service';
import { LinkedinServiceService } from './linkedin-service.service';
import { SoundcloudServiceService } from './soundcloud-service.service';
import { ModuleRef } from '@nestjs/core';

@Injectable()
export class ServiceFactory implements OnModuleInit {
  private servicesMap: Map<number, Type<BaseService>> = new Map();

  constructor(
    private elementService: ElementService,
    private eventEmitter: EventEmitter2,
    private serviceManager: ServiceManager,
    private moduleRef: ModuleRef,
  ) {}

  async onModuleInit(): Promise<any> {
    await this.initialize();
  }

  async initialize() {
    const services = await this.serviceManager.findAll();

    for (const service of services) {
      switch (service.name) {
        case 'google':
          this.servicesMap.set(service.id, GoogleServiceService);
          break;
        case 'microsoft':
          this.servicesMap.set(service.id, MicrosoftServiceService);
          break;
        case 'linkedin':
          this.servicesMap.set(service.id, LinkedinServiceService);
          break;
        case 'soundcloud':
          this.servicesMap.set(service.id, SoundcloudServiceService);
          break;
        default:
          throw new Error(`Unknown service: ${service.name}`);
      }
    }
  }

  async createService(serviceId: number): Promise<BaseService> {
    if (!this.servicesMap.has(serviceId)) {
      throw new Error(`Service with ID ${serviceId} not found`);
    }
    const ServiceClass = this.servicesMap.get(serviceId);
    if (!ServiceClass) {
      throw new Error(`Service class not found for serviceId ${serviceId}`);
    }
    return this.moduleRef.get(ServiceClass, { strict: false });
  }
}
